db.bills.find({_id:"user1"}, {"bills": { $elemMatch: {year: 2015}}, _id:0 }).pretty()


db.bills.aggregate([ 
	{ 
		$match: { _id:"user1" } 
	},
	{ 	
		$project: {
			bills: { 
				$filter: {
					input: "$bills",
					as: "bill",
					cond: { $eq: ["$$bill.year", 2015]}
				}
			},
			_id: 0
		}
	}
]).pretty()
